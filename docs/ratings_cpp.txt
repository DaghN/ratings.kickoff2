//Calculations when both players are rated
void BothRated(double OldRatingA, double OldRatingB, double &NewRatingA, double &NewRatingB, double ActualScore, double &ExpectedScore, double &RatingAdjustment)
{
	//Calculate expected score using Elo based on the logistic curve
	ExpectedScore = 1/(1+pow(10,(OldRatingB-OldRatingA)/400)); 

	//Calculate rating adjustment of team A based on K-factor, expected score, and actual score
	RatingAdjustment = Kfactor*(ActualScore-ExpectedScore); 

	//Adjust ratings
	NewRatingA = OldRatingA + RatingAdjustment;
	NewRatingB = OldRatingB - RatingAdjustment;
}
